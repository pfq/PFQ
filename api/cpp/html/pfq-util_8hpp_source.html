119  inline std::string
120  trim(std::string str)
121  {
122  auto b = str.find_first_not_of(" \n\r\t");
123  auto e = str.find_last_not_of(" \n\r\t");
124  b = b == std::string::npos ? 0 : b;
125  e = e == std::string::npos ? std::string::npos : (e + 1 - b);
126  return str.substr(b, e);
127  }
