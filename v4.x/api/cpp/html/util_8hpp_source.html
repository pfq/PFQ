201  inline uint32_t
202  symmetric_hash(const char *buf) noexcept
203  {
204  const char *ptr = buf;
205 
206  auto eh = reinterpret_cast<const ethhdr *>(ptr);
207  if (eh->h_proto != htons(0x800))
208  return 0;
209 
210  ptr += sizeof(ethhdr);
211 
212  auto ih = reinterpret_cast<const iphdr *>(ptr);
213  if (ih->protocol != IPPROTO_TCP &&
214  ih->protocol != IPPROTO_UDP)
215  return (ih->saddr ^ ih->daddr);
216 
217  ptr += sizeof(ih->ihl << 2);

219  auto uh = reinterpret_cast<const udphdr *>(ptr);
220  return (ih->saddr ^ ih->daddr ^ uh->source ^ uh->dest);

224  inline uint32_t
225  fold(uint32_t hash, size_t n) noexcept
226  {
227  hash = hash ^ (hash >> 8) ^ (hash >> 16) ^ (hash >> 24);
228 
229  switch(n)
230  {
231  case 1: return 0;
232  case 2: return hash & 1;
233  case 3: {
234  return (hash & 3) != 3 ? (hash & 3) : 0;
235  }
236  case 4: return hash & 3;
237  }

239  return hash % n;
