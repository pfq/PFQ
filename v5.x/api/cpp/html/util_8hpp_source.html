126  inline std::string
127  trim(std::string str)
128  {
129  auto b = str.find_first_not_of(" \n\r\t");
130  auto e = str.find_last_not_of(" \n\r\t");
131  b = b == std::string::npos ? 0 : b;
132  e = e == std::string::npos ? 0 : (e + 1 - b);
133  return str.substr(b, e);
134  }

233  inline uint32_t
234  symmetric_hash(const char *buf) noexcept
235  {
236  const char *ptr = buf;
237 
238  auto eh = reinterpret_cast<const ethhdr *>(ptr);
239  if (eh->h_proto != htons(0x800))
240  return 0;
241 
242  ptr += sizeof(ethhdr);
243 
244  auto ih = reinterpret_cast<const iphdr *>(ptr);
245  if (ih->protocol != IPPROTO_TCP &&
246  ih->protocol != IPPROTO_UDP)
247  return (ih->saddr ^ ih->daddr);
248 
249  ptr += sizeof(ih->ihl << 2);

251  auto uh = reinterpret_cast<const udphdr *>(ptr);
252  return (ih->saddr ^ ih->daddr ^ uh->source ^ uh->dest);
