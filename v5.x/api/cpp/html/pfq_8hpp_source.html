408  void
409  open(size_t caplen, size_t rx_slots, size_t tx_slots)
410  {
411  if (fd_ != -1)
412  throw pfq_error("PFQ: socket already open");
413 
414  fd_ = ::socket(PF_Q, SOCK_RAW, htons(ETH_P_ALL));
415  if (fd_ == -1)
416  throw pfq_error("PFQ: module not loaded");
417 
418  // allocate pdata
419 
420  data_.reset(new pfq_data { -1,
421  -1,
422  nullptr,
423  0,
424  nullptr,
425  0,
426  nullptr,
427  0,
428  0,
429  0,
430  0,
431  0,
432  0,
433  0,
434  0
435  });
436 
437  // get id
438 
439  data_->id = PFQ_VERSION_CODE;
440  socklen_t size = sizeof(data_->id);
441 
442  if (::getsockopt(fd_, PF_Q, Q_SO_GET_ID, &data_->id, &size) == -1)
443  throw pfq_error(errno, "PFQ: get id error");
444 
445  // set Rx queue slots
446 
447  if (::setsockopt(fd_, PF_Q, Q_SO_SET_RX_SLOTS, &rx_slots, sizeof(rx_slots)) == -1)
448  throw pfq_error(errno, "PFQ: set Rx slots error");
449 
450  data_->rx_slots = rx_slots;
451 
452  // set caplen
453  //
454 
455  if (::setsockopt(fd_, PF_Q, Q_SO_SET_RX_CAPLEN, &caplen, sizeof(caplen)) == -1)
456  throw pfq_error(errno, "PFQ: set Rx caplen error");
457 
458  data_->rx_slot_size = align<8>(sizeof(pfq_pkthdr) + caplen);
459 
460  // set Tx queue slots
461 
462  if (::setsockopt(fd_, PF_Q, Q_SO_SET_TX_SLOTS, &tx_slots, sizeof(tx_slots)) == -1)
463  throw pfq_error(errno, "PFQ: set Tx slots error");
464 
465 
466  // get maxlen

472  throw pfq_error(errno, "PFQ: get Tx maxlen error");

517  void
518  enable()
519  {
520  size_t tot_mem; socklen_t size = sizeof(tot_mem);
521 
522  if (data()->shm_addr != MAP_FAILED &&
523  data()->shm_addr != nullptr )
524  throw pfq_error(errno, "PFQ: queue already enabled");
525 
526  if (::getsockopt(fd_, PF_Q, Q_SO_GET_SHMEM_SIZE, &tot_mem, &size) == -1)
527  throw pfq_error(errno, "PFQ: queue memory error");
528 
529  auto env = getenv("PFQ_HUGEPAGES");
530  auto hugepages = hugepages_mountpoint();
531 
532  if (!hugepages.empty() &&
533  !getenv("PFQ_NO_HUGEPAGES") &&
534  (env == nullptr || atoi(env) != 0))
535  {
536  // HugePages
537  //
538  std::clog << "[PFQ] using HugePages..." << std::endl;
539 
540  hd_ = ::open((hugepages + "/pfq." + std::to_string(data_->id)).c_str(), O_CREAT | O_RDWR, 0755);
541  if (hd_ == -1)
542  throw pfq_error(errno, "PFQ: couldn't open a HugePages descriptor");
543 
544  data()->shm_addr = ::mmap(nullptr, tot_mem, PROT_READ|PROT_WRITE, MAP_SHARED, hd_, 0);
545  if (data()->shm_addr == MAP_FAILED)
546  throw pfq_error(errno, "PFQ: couldn't mmap HugePages");
547 
548  if(::setsockopt(fd_, PF_Q, Q_SO_ENABLE, &data()->shm_addr, sizeof(data()->shm_addr)) == -1)
549  throw pfq_error(errno, "PFQ: socket enable (HugePages)");
550  }
551  else
552  {
553  // standard pages (4K)
554  //
555 
556  std::clog << "[PFQ] using 4k-Pages..." << std::endl;
557 
558  void * null = nullptr;
559  if(::setsockopt(fd_, PF_Q, Q_SO_ENABLE, &null, sizeof(null)) == -1)
560  throw pfq_error(errno, "PFQ: socket enable");
561 
562  data()->shm_addr = ::mmap(nullptr, tot_mem, PROT_READ|PROT_WRITE, MAP_SHARED, fd_, 0);
563  if (data()->shm_addr == MAP_FAILED)
564  throw pfq_error(errno, "PFQ: socket enable (memory map)");
565  }
566 
567  data()->shm_size = tot_mem;
568 
569  data()->rx_queue_addr = static_cast<char *>(data()->shm_addr) + sizeof(pfq_shared_queue);
570  data()->rx_queue_size = data()->rx_slots * data()->rx_slot_size;
571 
572  data()->tx_queue_addr = static_cast<char *>(data()->shm_addr) + sizeof(pfq_shared_queue) + data()->rx_queue_size * 2;
573  data()->tx_queue_size = data()->tx_slots * data()->tx_slot_size;
574  }

1393  {

1395  throw pfq_error("PFQ: send_deferred: socket not enabled");

1398  data_->tx_num_bind);

1401 
1402  auto index = __atomic_load_n(&tx->cons, __ATOMIC_RELAXED);
1403  if (index != __atomic_load_n(&tx->prod, __ATOMIC_RELAXED))
1404  {

1409  //

1416  tx->ptr = base_addr;
1417  }

1420  //
