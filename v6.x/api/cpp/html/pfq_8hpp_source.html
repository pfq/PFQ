842  net_queue
843  read(long int microseconds = -1)
844  {
845  if (!data()->shm_addr)
846  throw system_error("PFQ: read: socket not enabled");
847 
848  auto q = static_cast<struct pfq_shared_queue *>(data()->shm_addr);
849  unsigned int data, index;
850 
851  data = __atomic_load_n(&q->rx.data, __ATOMIC_RELAXED);
852  index = Q_SHARED_QUEUE_INDEX(data);
853 
854  // at wrap-around reset Rx slots...
855  //
856 
857  if (((index+1) & 0xfe)== 0)
858  {
859  auto raw = static_cast<char *>(data_->rx_queue_addr) + ((index+1) & 1) * data_->rx_queue_size;
860  auto end = raw + data_->rx_queue_size;
861  const uint8_t rst = index & 1;
862  for(; raw < end; raw += data_->rx_slot_size)
863  reinterpret_cast<pfq_pkthdr *>(raw)->commit = rst;
864  }
865 
866  if (Q_SHARED_QUEUE_LEN(data) == 0)
867  {
868 #ifdef PFQ_USE_POLL
869  this->poll(microseconds);
870 #else
871  (void)microseconds;
872  return net_queue();
873 #endif
874  }
875 
876  // swap the net_queue...
877  //
878 
879  data = __atomic_exchange_n(&q->rx.data, (unsigned int)((index+1) << 24), __ATOMIC_RELAXED);
880 
881  auto queue_len = std::min(static_cast<size_t>(Q_SHARED_QUEUE_LEN(data)), data_->rx_slots);
882 
883  return net_queue(static_cast<char *>(data_->rx_queue_addr) + (index & 1) * data_->rx_queue_size,
884  data_->rx_slot_size, queue_len, index);
885  }

928  template <typename Fun>
929  size_t dispatch(Fun callback, long int microseconds = -1, char *user = nullptr)
930  {
931  auto many = this->read(microseconds);
932 
933  auto it = std::begin(many),
934  it_e = std::end(many);

937  {

939  std::this_thread::yield();

942  n++;

1135  }();

1138  //

1145  __atomic_store_n(&tx->prod.index, index, __ATOMIC_RELEASE);
1146  }

1149

1151  //

1153

1155  //

1157
